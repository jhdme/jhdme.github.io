    <button type="button" onclick="calculateProtein()">계산하기</button>

<p id="result"></p>


<script>

function calculateProtein() {

    const weight = document.getElementById('weight').value;

    const activityMultiplier = document.getElementById('activityLevel').value;

    const proteinIntake = weight * activityMultiplier;

    

    const resultElement = document.getElementById('result');

    resultElement.innerText = `하루 단백질 권장 섭취량: ${proteinIntake.toFixed(1)}g`;

}
